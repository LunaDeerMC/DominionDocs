```yaml :collapsed-lines
# この値は変更しないでください。
version: 2

# データベースの設定。
database:
  # サポートされているタイプ: sqlite, mysql, pgsql
  type: sqlite
  # データベースのホスト。
  host: localhost
  # データベースのポート。
  port: '3306'
  # データベース名。
  database: dominion
  # データベースのユーザー名。
  username: dominion
  # データベースのパスワード。
  password: dominion
  # データベースのコネクションプールサイズ。
  connection-pool-size: 10

# マルチサーバーの設定。
# 複数のサーバーがBungeeCordでプロキシされている場合、ここで設定して有効化できます。
# プレイヤーはマルチサーバー間で管理・テレポートできます。
# sqliteタイプのデータベースはマルチサーバーモードではサポートされていません。
# 既存データの移行についてはドキュメントを参照してください。
# https://dominion.lunadeer.cn/notes/doc/owner/other/multi-server/
multi-server:
  # マルチサーバーモードを有効化。
  enable: false
  # このサーバーの名前。
  # BC（Velocity）で設定したものと同じにしてください。
  server-name: server
  # このサーバーのID。他のサーバーと一意である必要があります。
  # 正の整数である必要があります。> 0
  # データベースにデータがある場合は変更しないでください。
  server-id: 1

# プラグインの言語。plugins/Dominion/languagesフォルダー内の他の言語も参照してください。
language: en_us

# 自動でドミニオンを作成する半径。-1で無効化。
auto-create-radius: 10

# プレイヤーがこの日数ログインしない場合、自動でドミニオンが削除されます。
# -1で無効化。
auto-clean-after-days: 180

# スポーン地点周辺でプレイヤーがドミニオンを作成するのを防止します。
server-spawn-protection-radius: 10

# ドミニオン作成時に位置選択に使うツール。
select-tool: ARROW

# クリックしたドミニオンの情報表示に使うツール。
info-tool: STRING

# プレイヤーのデフォルトUI設定。（Bedrockプレイヤーは常にCUIになります。）
# CUIまたはTUIに設定すると、プレイヤーはUIタイプを変更できません。
# BY_PLAYER: プレイヤー自身が決定。
# CUI: チェストGUI。
# TUI: テキストGUI。
default-ui-type: BY_PLAYER

# プラグインメッセージの設定。
plugin-message:
  # プレイヤーがドミニオンに入退場する際のデフォルトメッセージ。
  # 変数: {OWNER} - ドミニオンのオーナー, {DOM} - ドミニオン名
  default-enter-message: '&3{OWNER}: Welcome to {DOM}!'
  default-leave-message: '&3{OWNER}: Leaving {DOM}...'
  # メッセージの表示場所。
  # サポートされているタイプ: BOSS_BAR, ACTION_BAR, TITLE, SUBTITLE, CHAT
  no-permission-display-place: ACTION_BAR
  enter-leave-display-place: ACTION_BAR

# Webマップ上でドミニオンを描画します。
web-map-renderer:
  # サーバー起動後、全ドミニオンをWebマップに描画するまでの遅延秒数。
  # 描画前にワールドがロードされていることを確認してください。
  render-delay-sec: 40
  # https://bluemap.bluecolored.de/
  blue-map: false
  # https://www.spigotmc.org/resources/dynmap%C2%AE.274/
  dynmap: false
  # https://modrinth.com/plugin/squaremap
  square-map: false
  # https://modrinth.com/plugin/pl3xmap
  pl3x-map: false

# プレイヤーがレジデンスデータをドミニオンに移行できるかどうか。
residence-migration: false

# dominion.admin権限を持つプレイヤーがドミニオン制限をバイパスできるかどうか。
admin-bypass: true

# グループタイトルの設定。
# プレイヤーはグループ名をタブリストのタイトルとして使用できます。
group-title:
  enable: false
  prefix: '['
  suffix: ']'

# 外部リンクの設定。
external-links:
  command-help: ''
  documentation: https://dominion.lunadeer.cn/notes/doc/player/

# これらの権限ノードを持つプレイヤーはドミニオンの飛行制限の影響を受けません。
fly-permission-nodes:
  - essentials.fly
  - cmi.command.fly

# インターネット経由でアップデートを確認します。
check-update: true

# デバッグモード。バグ報告時はこれをオンにしてください。
debug: false

# パフォーマンスレコーダー。デバッグ時以外は開かないでください。
timer: false
```
