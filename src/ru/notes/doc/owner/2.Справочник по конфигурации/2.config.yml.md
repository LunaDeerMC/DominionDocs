```yaml :collapsed-lines
# Не изменяйте это значение.
version: 2

# Настройки базы данных.
database:
  # Поддерживаемые типы: sqlite, mysql, pgsql
  type: sqlite
  # Хост базы данных.
  host: localhost
  # Порт базы данных.
  port: '3306'
  # Имя базы данных.
  database: dominion
  # Имя пользователя базы данных.
  username: dominion
  # Пароль базы данных.
  password: dominion
  # Размер пула соединений с базой данных.
  connection-pool-size: 10

# Настройки мульти-сервера.
# Если у вас несколько серверов, проксированных через BungeeCord, вы можете настроить и включить это.
# Игрок может управлять/телепортироваться между серверами.
# База данных типа sqlite не поддерживается в режиме мульти-сервера.
# Для миграции существующих данных смотрите документацию.
# https://dominion.lunadeer.cn/notes/doc/owner/other/multi-server/
multi-server:
  # Включить режим мульти-сервера.
  enable: false
  # Имя этого сервера.
  # Должно совпадать с настройками в BC (Velocity).
  server-name: server
  # ID этого сервера, должен быть уникальным среди всех серверов.
  # Должен быть положительным целым числом. > 0
  # НЕ МЕНЯЙТЕ ЭТО ПОСЛЕ ТОГО, КАК В БАЗЕ ДАННЫХ ПОЯВЯТСЯ ДАННЫЕ.
  server-id: 1

# Язык плагина, другие смотрите в папке plugins/Dominion/languages.
language: en_us

# Радиус автоматического создания доминиона. -1 для отключения.
auto-create-radius: 10

# Если игрок не заходит в течение этого количества дней, его доминион будет автоматически удалён.
# Установите -1 для отключения.
auto-clean-after-days: 180

# Запретить игроку создавать доминион вокруг точки спауна.
server-spawn-protection-radius: 10

# Инструмент для выбора позиции при создании доминиона.
select-tool: ARROW

# Инструмент для отображения информации о выбранном доминионе.
info-tool: STRING

# Настройки типа интерфейса игрока по умолчанию. (Для bedrock-игрока всегда будет CUI.)
# Если установлено CUI или TUI, игрок не может изменить тип интерфейса.
# BY_PLAYER: Решает сам игрок.
# CUI: Chest GUI.
# TUI: Text GUI.
default-ui-type: BY_PLAYER

# Настройки сообщений плагина.
plugin-message:
  # Сообщение по умолчанию при входе/выходе игрока из доминиона.
  # Переменные: {OWNER} - владелец доминиона, {DOM} - имя доминиона.
  default-enter-message: '&3{OWNER}: Welcome to {DOM}!'
  default-leave-message: '&3{OWNER}: Leaving {DOM}...'
  # Где показывать сообщение.
  # Поддерживаемые типы: BOSS_BAR, ACTION_BAR, TITLE, SUBTITLE, CHAT
  no-permission-display-place: ACTION_BAR
  enter-leave-display-place: ACTION_BAR

# Отображение доминиона на веб-карте.
web-map-renderer:
  # Задержка в секундах после запуска сервера для отображения всех доминионов на веб-карте.
  # Убедитесь, что миры загружены перед рендерингом.
  render-delay-sec: 40
  # https://bluemap.bluecolored.de/
  blue-map: false
  # https://www.spigotmc.org/resources/dynmap%C2%AE.274/
  dynmap: false
  # https://modrinth.com/plugin/squaremap
  square-map: false
  # https://modrinth.com/plugin/pl3xmap
  pl3x-map: false

# Можно ли игроку мигрировать данные residence в доминион.
residence-migration: false

# Может ли игрок с правом dominion.admin обходить ограничения доминиона.
admin-bypass: true

# Настройки группового титула.
# Игрок может использовать имя своей группы как титул в списке игроков.
group-title:
  enable: false
  prefix: '['
  suffix: ']'

# Настройки внешних ссылок.
external-links:
  command-help: ''
  documentation: https://dominion.lunadeer.cn/notes/doc/player/

# Игроки с этими правами не будут ограничены лимитом полёта доминиона.
fly-permission-nodes:
  - essentials.fly
  - cmi.command.fly

# Проверять обновления через интернет.
check-update: true

# Режим отладки, включите при отправке баг-репортов.
debug: false

# Рекордер производительности, не включайте, если не отлаживаете.
timer: false
```
